script
var gulp = require('gulp');
var stylus = require('gulp-stylus');

gulp.task('styles', function() {
	// body...
	gulp.src('styles.styl')
		.pipe(stylus())
		.pipe(gulp.dest('./'));
});

gulp.task('watch:styles', function(){
	gulp.watch('**/*.styl', ['styles']);
});
